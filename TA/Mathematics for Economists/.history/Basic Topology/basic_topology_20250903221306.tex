    {\small  % Make text smaller to fit columns better  
        \begin{columns}[T]
            \begin{column}{0.48\textwidth}
                Left column content: \
                Equations look like: \
                $E=mc^2$ \ [2ex]  
                
                \includegraphics[width=\linewidth]{example-image-a}   
            \end{column}\hfill  
            \begin {column }{0 .48\textwidth }  
             Right column content: \
             More text here or another graphic \
             Or a table: \ [1ex ]   
             
            \begin{tabular}{|c|c|}  
              A & B \  	
              C & D \  
            End{tabular }   
         End {column }   
      End {columns }     
   }
